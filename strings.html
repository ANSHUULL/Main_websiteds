      #include<conio.h>
      #include<dos.h>

      void main()
{

   char str[80];
   int i;
   clrscr();

   printf("Enter any string:- ");
   gets(str);

   for (i = 0; str[i] != '\0'; i++)
   {
      if (str[i] >= 'A' && str[i] <= 'Z')
      {
         str[i] = str[i] + 32;
      }
   }
   str[0] = str[0] - 32;

   for (i = 1; str[i] != '0'; i++)
   {

      if (str[i] == ' ')
         str[i + 1] = str[i + 1] - 32;
   }
   printf("String in title case:- %s", str);

   delay(9000);
} 
         
    
    
    
              
Output:- 

                
              Enter any string:- the portrait lady
              String in title case:- The Portrait Lady

 8) PROGRAM :- ENTER TWO STRINGS FROM USER AND CONCATENATE THOSE STRINGS IN ANOTHER STRING AND DISPLAY THE ANSWER 

       
      #include<stdio.h> 

      int main()
{
   char s1[100] = "programming ", s2[] = "is awesome";
   int length, j;

   // store length of s1 in the length variable
   length = 0;
   while (s1[length] != '\0')
   {
      ++length;
   }

   // concatenate s2 to s1
   for (j = 0; s2[j] != '\0'; ++j, ++length)
   {
      s1[length] = s2[j];
   }

   // terminating the s1 string
   s1[length] = '\0';

   printf("After concatenation: ");
   puts(s1);

   return 0;
}

     

    
    
              
Output:- 

                
              After concatenation: programming is awesome

 10) PROGRAM :- ENTER THE STRING FROM USER, ENTER THE INDEX NUMBER AND PRINT CHARACTER PRESENT ON THAT INDEX ? 

       
      #include<stdio.h>

      int main()
{
   char inputString[100];
   int index;

   printf("Enter a string: ");
   scanf("%99s", inputString);

   printf("Enter the index: ");
   scanf("%d", &index);

   int stringLength = 0;

   while (inputString[stringLength] != '\0')
   {
      stringLength++;
   }

   if (index >= 0 && index < stringLength)
   {
      printf("Character at index %d: %c\n", index, inputString[index]);
   }
   else
   {
      printf("Invalid index. The string has %d characters.\n", stringLength);
   }

   return 0;
}

        
    
    
              
Output:- 

               
              Enter a string: hello
              Enter the index: 2
              Character at index 2: l
              
          
      
 11) PROGRAM :- ENTER THE STRING FROM USER AND CHECK ITS PALINDROME OR NOT? 

       
      #include<stdio.h>

      int isPalindrome(char *str)
{
    int left = 0;
    int right = 0;

    
    while (str[right] != '\0')
    {
        right++;
    }
    right--;

    
    while (left < right)
    {
        if (str[left] != str[right])
        {
            return 0; 
        }
        left++;
        right--;
    }
    return 1; 
}

int main()
{
    char str[100];

    printf("Enter a string: ");
    scanf("%s", str);

    if (isPalindrome(str))
    {
        printf("%s is a palindrome.\n", str);
    }
    else
    {
        printf("%s is not a palindrome.\n", str);
    }

    return 0;
}

        
    
    
              
Output:- 

               
              Enter a string: nitin
              nitin is a palindrome.

 13) PROGRAM :- ENTER THE STRING FROM THE USER AND COPY REVERSE OF THE STRING INTO ANOTHER STRING? 

       
      #include<stdio.h>

      int main()
{
    char str[100];
    char reverseStr[100];

    printf("Enter a string: ");
    scanf("%s", str);

    
    int length = 0;
    while (str[length] != '\0')
    {
        length++;
    }

    
    int i, j;
    for (i = length - 1, j = 0; i >= 0; i--, j++)
    {
        reverseStr[j] = str[i];
    }
    reverseStr[j] = '\0'; 

    printf("Original string: %s\n", str);
    printf("Reversed string: %s\n", reverseStr);

    return 0;
}

        
    
    
              
Output:- 

               
              Enter a string: hello
              Original string: hello
              Reversed string: olleh
